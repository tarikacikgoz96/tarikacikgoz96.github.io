ABAP REST API Send HTTP Request (GET, POST etc.)

Published: 29.06.2026 Updated: 29.06.2026

ABAP programı içerisinden HTTP isteği gönderip cevap almak içindir.

Dönen JSON response (string) sonucunun işlenmesine başka bir başlıkta değinilecektir.

Kodda “for debug” olan kısımlar ile bağlantı problemi yaşarsanız, daha detaylı bilgi alabilirsiniz.

İsterseniz tek bir sorgu ihtiyacı için kullanın, isterseniz size gelen RESTful API paketinin tamamını bu metodu kullanarak implemente edin.

Dikkat: Eğer sorgu sonucunda bağlantı hatası vb. beklenmeyen sonuçlar alıyorsanız, lütfen STRUST tcode kullanarak ilgili REST API servis sağlayıcısının sertifikasını yüklemeyi unutmayınız. Size sorgudan dönen sonuçta, doğrudan sertifika hatası demese de SAP ortamında API sorgularında yaşanan bağlantı hatalarının büyük çoğunluğu sertifika yüklenmemesinden kaynaklıdır.

Temel çalıştırma aşağıdaki gibidir;

*&---------------------------------------------------------------------*
*& Report ZTEMPLATE_RESTAPI
*&---------------------------------------------------------------------*
*&
*&---------------------------------------------------------------------*
REPORT ZTEMPLATE_RESTAPI.


START-OF-SELECTION.


  data: lv_method TYPE string,
        lv_url TYPE string,
        lt_headers TYPE TIHTTPNVP,
        ls_header like LINE OF lt_headers,
        lv_body TYPE string,

        lv_response TYPE string,
        lv_http_status TYPE string,
        lv_http_reason type string,
        lv_message TYPE string.

  DATA: lo_cls TYPE REF TO zcl_template.

  CREATE OBJECT lo_cls.





  lv_method = 'POST'.
  lv_url = 'https://reqbin.com/echo/post/json'.

  clear ls_header.
  ls_header-name = 'Accept'.
  ls_header-value = '	application/json'.
  APPEND ls_header to lt_headers.

  lv_body = '{'  &&
  '"Id": 78912,' &&
  '"Customer": "Jason Sweet",' &&
  '"Quantity": 1,' &&
  '"Price": 18.00' &&
  '}'.





  lo_cls->send_http_request(
    EXPORTING
      iv_method        = lv_method
      iv_url           = lv_url
      it_headers       = lt_headers
      iv_body          = lv_body
*      iv_proxy_host    =
*      iv_proxy_service =
*      iv_basic_auth    =
*      iv_username      =
*      iv_password      =
    IMPORTING
      ev_response      = lv_response
      ev_http_status   = lv_http_status
      ev_http_reason   = lv_http_reason
      ev_message       = lv_message
  ).


  BREAK-POINT.



END-OF-SELECTION.

SEND_HTTP_REQUEST metodu;

*Method Signature:
*IV_METHOD      Importing	Type	STRING
*IV_URL         Importing	Type	STRING
*IT_HEADERS     Importing	Type	TIHTTPNVP
*IV_BODY        Importing	Type	STRING
*IV_PROXY_HOST  Importing	Type	STRING      (optinal)
*IV_PROXY_SERVICE Importing	Type	STRING      (optinal)
*IV_BASIC_AUTH	Importing	Type	FLAG        (optinal)
*IV_USERNAME	Importing	Type	STRING      (optinal)
*IV_PASSWORD	Importing	Type	STRING      (optinal)
*EV_RESPONSE    Exporting	Type	STRING      (optinal)
*EV_HTTP_STATUS	Exporting	Type	STRING
*EV_HTTP_REASON	Exporting	Type	STRING
*EV_MESSAGE     Exporting	Type	STRING

  METHOD SEND_HTTP_REQUEST.

    DATA:
      lo_http_client           TYPE REF TO if_http_client,
      lo_rest_client           TYPE REF TO cl_rest_http_client,
      lo_response              TYPE REF TO if_rest_entity.

    ev_message = 'OK'.

    TRY.

        CALL METHOD cl_http_client=>create_by_url
          EXPORTING
            url                = iv_url
            proxy_host         = iv_proxy_host
            proxy_service      = iv_proxy_service
          IMPORTING
            client             = lo_http_client
          EXCEPTIONS
            argument_not_found = 1
            plugin_not_active  = 2
            internal_error     = 3
            OTHERS             = 5.
        IF sy-subrc NE 0.
          ev_message = 'Http isteği oluşturulamadı.'.
          RETURN.
        ENDIF.

        "set headers------------
*        lo_http_client->request->set_header_field(
*          name = 'Content-Type'
*          value = 'application/json'
*          ).
        "...
        "set headers------------

        IF iv_basic_auth eq 'X'.
          lo_http_client->AUTHENTICATE(
            exporting
*              PROXY_AUTHENTICATION = ' '    " Proxy Logon (= 'X')
*              CLIENT               =     " R/3 System, client number from logon
              USERNAME             =  IV_USERNAME   " ABAP System, User Logon Name
              PASSWORD             =  IV_PASSWORD  " Logon ID
*              LANGUAGE             =     " SAP System, Current Language
          ).
        ENDIF.

        lo_http_client->request->set_header_fields( fields = it_headers ).
        lo_http_client->request->set_method( iv_method ).
      CATCH cx_root INTO DATA(roott).
        ev_message =  roott->get_longtext( ).
        RETURN.
    ENDTRY.


    TRY.

        CREATE OBJECT lo_rest_client
          EXPORTING
            io_http_client = lo_http_client.

        lo_http_client->request->set_cdata(
          EXPORTING
            data   =     iv_body
        ).


        lo_http_client->send( ).
        lo_http_client->receive(
          EXCEPTIONS
            http_communication_failure = 1
            http_invalid_state         = 2
            http_processing_failed     = 3
            others                     = 4
        ).
        IF sy-subrc NE 0.
          "for debug--------
          data:
          lv_CODE Type  SYSUBRC,
          lv_MESSAGE Type	STRING,
          lv_MESSAGE_CLASS Type	ARBGB,
          lv_MESSAGE_NUMBER Type  MSGNR.
          lo_http_client->GET_LAST_ERROR(
            importing
              CODE           = lv_CODE
              MESSAGE        = lv_MESSAGE
              MESSAGE_CLASS  = lv_MESSAGE_CLASS
              MESSAGE_NUMBER = lv_MESSAGE_NUMBER
          ).
          ev_message = 'NOK'.
          "for debug--------
        ENDIF.

        lo_response = lo_rest_client->if_rest_client~get_response_entity( ) .
        "for debug--------
        data et_status type TIHTTPNVP.
        et_status = lo_response->get_header_fields( ).
        "for debug--------
        ev_http_reason = lo_response->get_header_field( '~status_reason' ).
        ev_http_status = lo_response->get_header_field( '~status_code' ).
        IF ev_http_status ne '200'.
          ev_message = 'NOK'.
        ENDIF.
        ev_response = lo_response->get_string_data( ).


        IF lo_rest_client IS BOUND.
          lo_http_client->refresh_response( ).
          lo_http_client->close( ).
        ENDIF.


    CATCH cx_root INTO DATA(root).
      ev_message =  root->get_longtext( ).
      RETURN.
    ENDTRY.


  ENDMETHOD.
